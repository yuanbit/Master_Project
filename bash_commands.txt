for img in *.jpg; do
    filename=${img%.*}
    convert "$filename.jpg" "$filename.png"
done

for f in *; do mv "$f" "${f// /_}"; done

rename "s/ /_/g" *

for file in *; do mv "$file" $(echo "$file" | sed -e 's/[^A-Za-z0-9._-]/_/g'); done &

rename 's/\.png$/_golf.png/' *png

for x in ./*.png; do
  mkdir "${x%.*}" && mv "$x" "${x%.*}"
done

 
for i in *png
do
    mv "$i" "`echo $i | sed 's/000//'`"
done
